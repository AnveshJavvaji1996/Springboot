import psutil
import time
from datetime import datetime

# Check every 5 hours (5 × 60 × 60 seconds)
INTERVAL = 5 * 60 * 60

while True:
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    cpu_usage = psutil.cpu_percent(interval=1)

    print(f"[{timestamp}] CPU Usage: {cpu_usage}%")

    time.sleep(INTERVAL)
    time.sleep(INTERVAL)
